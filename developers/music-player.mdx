## Offline Mode

### Audio Caching with IndexedDB

```typescript title="site/src/lib/audio-cache.ts" lines
const DB_NAME = 'rotations-audio-cache';
const STORE_NAME = 'audio-files';

export class AudioCache {
  private db: IDBDatabase | null = null;
  
  async init() {
    this.db = await openDB(DB_NAME, 1, {
      upgrade(db) {
        db.createObjectStore(STORE_NAME);
      }
    });
  }
  
  async cacheTrack(trackId: string, audioBlob: Blob) {
    await this.db.put(STORE_NAME, audioBlob, trackId);
  }
  
  async getCachedTrack(trackId: string): Promise<Blob | null> {
    return await this.db.get(STORE_NAME, trackId);
  }
  
  async isCached(trackId: string): Promise<boolean> {
    const count = await this.db.count(STORE_NAME, trackId);
    return count > 0;
  }
}
```

### Cache-First Playback

```typescript title="site/src/hooks/usePlayer.ts" lines
export function usePlayer() {
  const audioCache = useRef(new AudioCache());
  
  async function playTrack(trackId: string) {
    const track = store.getCell('tracks', trackId, 'url');
    
    // Try cache first
    const cached = await audioCache.current.getCachedTrack(trackId);
    if (cached) {
      const url = URL.createObjectURL(cached);
      audioRef.current.src = url;
      audioRef.current.play();
      return;
    }
    
    // Stream and cache in background
    audioRef.current.src = track;
    audioRef.current.play();
    
    // Cache for next time
    fetch(track)
      .then(r => r.blob())
      .then(blob => audioCache.current.cacheTrack(trackId, blob));
  }
}
```

## Rotation System

### Current Rotation Tracking

```typescript title="site/src/hooks/useRotation.ts" lines
import { store } from '../lib/tinybase-store';

export function useRotation() {
  const [rotation, setRotation] = useState<string[]>([]);
  
  // Add to rotation when a track is played
  useEffect(() => {
    const listener = store.addListener(
      'currentTrack',
      null,
      (store, tableId, rowId) => {
        const trackId = rowId as string;
        if (!rotation.includes(trackId)) {
          setRotation(prev => [...prev, trackId]);
        }
      }
    );
    
    return () => store.delListener(listener);
  }, [rotation]);
  
  return { rotation, setRotation };
}
```

### Rotation Radius

Control how many songs outside your rotation to show:

```typescript
interface RotationConfig {
  radius: number;        // How many songs outside rotation to include
  algorithm: 'random' | 'pagerank' | 'recent';
}

function getRecommendations(rotation: string[], config: RotationConfig) {
  const inRotation = new Set(rotation);
  const allTracks = store.getTable('tracks');
  
  // Get tracks outside rotation
  const candidates = Object.keys(allTracks)
    .filter(id => !inRotation.has(id));
  
  // Apply algorithm
  switch (config.algorithm) {
    case 'random':
      return shuffle(candidates).slice(0, config.radius);
    case 'pagerank':
      return candidates
        .sort((a, b) => getPageRank(b) - getPageRank(a))
        .slice(0, config.radius);
    case 'recent':
      return candidates
        .sort((a, b) => getLastPlayed(b) - getLastPlayed(a))
        .slice(0, config.radius);
  }
}
```

## Playlist Management

### Creating Playlists

```typescript title="site/src/components/PlaylistManager.tsx" lines
function createPlaylist(name: string, trackIds: string[]) {
  const playlistId = `playlist-${Date.now()}`;
  
  // Add playlist
  store.setRow('playlists', playlistId, {
    id: playlistId,
    name,
    createdAt: Date.now()
  });
  
  // Add tracks
  trackIds.forEach((trackId, index) => {
    store.setRow('playlistTracks', `${playlistId}-${trackId}`, {
      playlistId,
      trackId,
      position: index
    });
  });
}
```

### Cross-Playlist Next Tracks

Track next-track relationships across playlists:

```typescript
interface Track {
  next_tracks?: Record<PlaylistId, TrackId>;
}

// When a track ends in a playlist context
function getNextTrack(currentTrackId: TrackId, playlistId: PlaylistId) {
  const track = store.getRow('tracks', currentTrackId);
  
  // Check explicit next track for this playlist
  if (track.next_tracks?.[playlistId]) {
    return track.next_tracks[playlistId];
  }
  
  // Fall back to next in playlist order
  const playlistTracks = store
    .getRowsWhere('playlistTracks', row => row.playlistId === playlistId)
    .sort((a, b) => a.position - b.position);
  
  const currentIndex = playlistTracks.findIndex(t => t.trackId === currentTrackId);
  return playlistTracks[currentIndex + 1]?.trackId;
}
```
