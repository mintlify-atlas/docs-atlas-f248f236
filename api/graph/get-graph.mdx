title: Get Graph

---

## Endpoint

```
GET /graph
```

Returns the complete listening graph for a user, including all nodes with their relationships, metadata, and analysis results. Supports optional pagination for large graphs.

## Query Parameters

<ParamField query="user" type="string" required>
  The Last.fm username whose graph to retrieve
</ParamField>

<ParamField query="limit" type="integer" default="0">
  Maximum number of nodes to return. When set to 0 or omitted, returns all nodes.
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of nodes to skip (for pagination). Only used when `limit` is set.
</ParamField>

## Response

<ResponseField name="nodes" type="object">
  Dictionary of nodes keyed by UUID
  
  <Expandable title="CompactGraphNode properties">
    <ResponseField name="songKey" type="string">
      Canonical song identifier in format `artist::track` (both lowercase)
    </ResponseField>
    
    <ResponseField name="mbid" type="string">
      MusicBrainz ID from Last.fm (if available)
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Track name
    </ResponseField>
    
    <ResponseField name="artists" type="string[]">
      List of artist names
    </ResponseField>
    
    <ResponseField name="albumName" type="string">
      Album name (if available)
    </ResponseField>
    
    <ResponseField name="lastfmUrl" type="string">
      Last.fm URL for the track
    </ResponseField>
    
    <ResponseField name="imageUrl" type="string">
      Album artwork URL
    </ResponseField>
    
    <ResponseField name="next" type="object">
      Outgoing edges: UUID → edge weight (number of times this song was followed by that song)
    </ResponseField>
    
    <ResponseField name="previous" type="object">
      Incoming edges: UUID → edge weight (number of times this song followed that song)
    </ResponseField>
    
    <ResponseField name="totalPlays" type="integer">
      Total number of times this song was played
    </ResponseField>
    
    <ResponseField name="sources" type="string[]">
      Data sources (currently only `["lastfm"]`)
    </ResponseField>
    
    <ResponseField name="sourcePlays" type="object">
      Breakdown of plays per source: `{ "lastfm": number }`
    </ResponseField>
    
    <ResponseField name="pageRank" type="number">
      PageRank score (centrality measure)
    </ResponseField>
    
    <ResponseField name="clusterId" type="integer">
      Community cluster ID assigned by Louvain algorithm
    </ResponseField>
    
    <ResponseField name="playDates" type="string[]">
      ISO 8601 timestamps of all plays, chronologically sorted
    </ResponseField>
    
    <ResponseField name="positions" type="object">
      Pre-computed layout positions for visualization
      
      <Expandable title="Layout positions">
        <ResponseField name="pagerank" type="object">
          Position in PageRank-based layout: `{ x: number, y: number }`
        </ResponseField>
        <ResponseField name="mds" type="object">
          Position in MDS layout: `{ x: number, y: number }`
        </ResponseField>
        <ResponseField name="weighted-mds" type="object">
          Position in weighted MDS layout: `{ x: number, y: number }`
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="metadata" type="object">
  Graph metadata
  
  <Expandable title="GraphMetadata properties">
    <ResponseField name="totalScrobbles" type="integer">
      Total number of scrobbles used to build this graph
    </ResponseField>
    
    <ResponseField name="dateRange" type="object">
      Time range of scrobble data
      <Expandable>
        <ResponseField name="from" type="string">
          ISO 8601 timestamp of earliest scrobble
        </ResponseField>
        <ResponseField name="to" type="string">
          ISO 8601 timestamp of latest scrobble
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="exportTimestamp" type="string">
      ISO 8601 timestamp when this graph was built
    </ResponseField>
    
    <ResponseField name="lastfmUsername" type="string">
      Last.fm username
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object">
  Pagination information (only present when `limit` is set)
  
  <Expandable title="Pagination properties">
    <ResponseField name="total" type="integer">
      Total number of nodes in the graph
    </ResponseField>
    
    <ResponseField name="offset" type="integer">
      Current offset
    </ResponseField>
    
    <ResponseField name="limit" type="integer">
      Current limit
    </ResponseField>
    
    <ResponseField name="hasMore" type="boolean">
      Whether more nodes are available
    </ResponseField>
  </Expandable>
</ResponseField>

## Notes

<Note>
- Node UUIDs are stable for a given user's graph
- The `next` and `previous` fields use UUIDs for neighbor references (not SongKeys)
- Pagination preserves the full `metadata` object in each response
- Large graphs (>1000 nodes) should use pagination for better performance
</Note>
